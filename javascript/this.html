this绑定规则

this指向大概有5中规则(规则的权重大小是逐渐增大的，当可以应用多个规则时取权重大的)：

  • 默认绑定
  • 隐含绑定
  • 明确绑定
  • new绑定
  • 箭头函数绑定

默认绑定

独立函数调用。可以认为这种 this 规则是在没有其他规则适用时的默认规则。
比如：

// 根据this来访问变量
function foo() {
  var a = 'foo'
	console.log( this.a );
}
var a = 'window'
foo() // window

隐含绑定

function foo() {
  console.log(this.a)
}
var obj = {
  a: 'obj',
  foo,
}
var a = 'window'
obj.foo() // obj

// 或者
var obj = {
  a: 'obj',
  foo: function foo() {
    console.log(this.a)
  }
}
var a = 'window'
obj.foo() // obj

明确绑定

通过调用 call 、 apply 、 bind 函数或者给JS本身某些内置方法的可选this参数传递值，来明确的绑定this。

function bar() {
  console.log(this.a)
}
var obj1 = {
  a: 'obj1'
}
var obj2 = {
  a: 'obj2'
}
var obj3 = {
  a: 'obj3'
}
bar.call(obj1) // obj1
bar.apply(obj2) // obj2
bar() // undefined
var baz = bar.bind(obj3)
baz() // obj3

箭头函数绑定

function foo() {
  // 因为foo是通过call来调用的，所以此时foo的this是指向{ id: 42 }这个对象的
  // 箭头函数强行绑定了setTimeout里面函数的this就是外部环境的this
  setTimeout(() => {
    console.log('id:', this.id);
  }, 100);
}

var id = 21;
// 隐含帮绑定
foo.call({ id: 42 }); // 42



function foo1() {
  // setTimeout里面的函数是独立调用，适用第一条规则默认绑定，this是window对象
  setTimeout(function() {
    console.log('id:', this.id);
  }, 100);
}

var id = 21;

foo1.call({ id: 42 }); // 21

箭头函数有几个需要注意的点：

  • 函数体内的this对象，就是定义时所在的对象，而不是使用时所在的对象。
  • 不可以使用arguments对象
  • 不能使用 new 操作符
  • 不可以使用yield命令

## 手动实现call、apply、bind #### call 
Function.prototype.call = function() {
  let context = arguments[0]
  if (context === undefined || context === null) {
    context = window
  }
  context.fn = this
  if (typeof context.fn !== 'function') {
    throw new TypeError('请用函数调用call方法')
  }
  const args = [...arguments].slice(1)
  const result = context.fn(...args)
  delete context.fn
  return result
}

apply

Function.prototype.apply = function(context, args) {
  if (context === undefined || context === null) {
    context = window
  }
  context.fn = this
  if (typeof context.fn !== 'function') {
    throw new TypeError('请用函数调用apply方法')
  }
  const result = args ? context.fn(...args) : context.fn()
  delete context.fn
  return result
}

bind

Function.prototype.bind = function() {
  const fn = this
  if (typeof fn !== 'function') {
    throw new TypeError('请用函数调用bind方法')
  }
  const context = arguments[0]
  const args = Array.prototype.slice.call(arguments, 1)
  return function bindFn() {
    let args1 = args.concat(Array.prototype.slice.call(arguments))
    if (this instanceof bindFn) {
      return fn.apply(this, args1)
    } else {
      return fn.apply(context, args1)
    }
  }
}
