# 动态规划
**动态规划**（英语：Dynamic programming，简称DP）是一种在数学、管理科学、计算机科学、经济学和生物信息学中使用的，通过把原问题分解为相对简单的子问题的方式求解复杂问题的方法。  ——维基百科

### 1. 连续子数组的最大和（剑指 Offer-42）

<br />**解法：**<br />
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
  const len = nums.length
  if (!len) return 0
  let max = nums[0]
  let sum = nums[0]
  for (let i = 1; i < len; i++) {
    if (sum > 0) {
      sum += nums[i]
    } else {
      sum = nums[i]
    }
    max = Math.max(max, sum)
  }
  return max
};
```

function climbStairs(n: number): number {
  let p: number = 0, q: number = 0, r: number = 1;
  for (let i = 1; i <= n; ++i) {
    p = q; 
    q = r; 
    r = p + q;
  }
  return r;
};

### 3. 使用最小花费爬楼梯
![image.png](https://cdn.nlark.com/yuque/0/2020/png/299895/1595774792397-56155893-d6d7-4c62-9a64-0e2059f015f7.png#align=left&display=inline&height=273&margin=%5Bobject%20Object%5D&name=image.png&originHeight=546&originWidth=916&size=57035&status=done&style=none&width=458)<br />**解法一：从前往后**
```typescript
// f[i] = cost[i] + min(f[i-1], f[i-2])
function minCostClimbingStairs(cost: number[]): number {
  let temp: number = 0
  // 前一个的花费
  let pre1:number = cost[1]
  // 前两个的花费
  let pre2: number = cost[0]
  for (let i = 2; i < cost.length; i++) {
    temp = Math.min(pre1, pre2) + cost[i]
    pre2 = pre1
    pre1 = temp
  }
  return Math.min(pre1, pre2)
};
```
**解法二：从后往前**
```typescript
function minCostClimbingStairs(cost: number[]): number {
  let f1: number = 0
  let f2: number = 0
  for (let i = cost.length - 1; i >= 0; --i) {
    let temp = cost[i] + Math.min(f1, f2)
    f2 = f1
    f1 = temp
  }
  return Math.min(f1, f2)
};
```

### 4. 三步问题（面试题08.01）
![image.png](https://cdn.nlark.com/yuque/0/2020/png/299895/1595776813102-d7c4fc76-70ff-4112-b9db-bbc5a4ebacee.png#align=left&display=inline&height=220&margin=%5Bobject%20Object%5D&name=image.png&originHeight=439&originWidth=923&size=30957&status=done&style=stroke&width=461.5)
```typescript
function waysToStep(n: number): number {
  if (n === 1) return 1
  if (n === 2) return 2
  if (n === 3) return 4
  let ans:number = 0
  let i:number = 4
  // 前一位
  let pre1:number = 4
  // 前两位
  let pre2:number = 2
  // 前三位
  let pre3:number = 1
  while (i <= n) {
    ans = (pre1 + pre2 + pre3) % 1000000007
    pre3 = pre2
    pre2 = pre1
    pre1 = ans
    i++
  }
  return ans 
};
```

### 5. 打家劫舍（LeetCode-198）
![image.png](https://cdn.nlark.com/yuque/0/2020/png/299895/1595816158554-3e549f5f-6cce-4be9-9c12-b2976f497f5e.png#align=left&display=inline&height=567&margin=%5Bobject%20Object%5D&name=image.png&originHeight=567&originWidth=598&size=35831&status=done&style=none&width=598)<br />**解法一：**
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function(nums) {
  const len = nums.length
  let result = [0]
  for (let i = 0; i < len; i++) {
    if (i >= 2) {
      result.push(Math.max.apply(null, result.slice(0, i)) + nums[i])
    } else {
      result.push(nums[i])
    }
  }
  return Math.max.apply(null, result)
};
```

  const len = nums.length
  if (len < 3)return Math.max.apply(null, [...nums, 0])

  // 前一位

  }

/**

var maxSubArray = function(nums) {
  const len = nums.length
  let result = []
  for (let i = 0; i < len; i++) {
    if (i >= 1) {
      
      result.push(Math.max(result[i - 1] + nums[i], nums[i]))
    } else {
      result.push(nums[i])
    }
  }
  return Math.max(...result)
};

 * @param {number[]} nums

var maxSubArray = function(nums) {
  const len = nums.length
  // 赋值-Infinity保证初始值为最小
  let ans = -Infinity
  let pre = -Infinity
  for (let i = 0; i < len; i++) {
    pre = Math.max(pre + nums[i], nums[i])
    ans = Math.max(ans, pre)
  }

var maxProfit = function(prices) {
  const len = prices.length
  let ans = 0
  prices.forEach((val, index) => {
    ans = Math.max(ans, Math.max.apply(null, prices.slice(index + 1)) - val)
  })
  return ans

/**
 * @param {number[]} prices

var maxProfit = function(prices) {
  let min = prices[0]
  let res = 0
  prices.forEach(val => {
    res = Math.max(res, val - min)
    // 保证min是最小值

  })

### 8. 除数博弈（LeetCode-1025）
![image.png](https://cdn.nlark.com/yuque/0/2020/png/299895/1595821514930-cd7712cd-11a9-42ab-a82c-26a6d98a10b3.png#align=left&display=inline&height=580&margin=%5Bobject%20Object%5D&name=image.png&originHeight=580&originWidth=630&size=33222&status=done&style=none&width=630)<br />**解法一：**
```javascript
/**
 * @param {number} N
 * @return {boolean}
 */
var divisorGame = function(N) {
  let result = []
  result[1] = false
  result[2] = true
  for (let i = 3; i <= N; i++) {
    // 判断当 N = i时的胜负情况
    for (let j = 1; j < i; j++) {
      // 如果i % j === 0 并且当 N = i - j的时候，爱丽丝输了，则爱丽丝必胜
      if ( i % j === 0 && !result[i - j]) {
        result[i] = true
        break
      } else {
        result[i] = false
      }
    }
    
  }
  return result[N]
};
```
