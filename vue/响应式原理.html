响应式原理

在Vue 初始化的过程中，会触发initState 方法初始化props、 methods、data 、computed、watch等属性。
在初始化data 的最后会执行：

observe(data, true / asRootData /)
1

来把data 变成响应式的数据。

observe

返回一个 Observer 实例：

 function observe (value: any, asRootData: ?boolean): Observer | void {
   // 如果value不是对象或者value是VNode的实例
   if (!isObject(value) || value instanceof VNode) {
     return
   }
   let ob: Observer | void
   // 如果value已经是响应式数据直接返回
   if (hasOwn(value, '__ob__') && value.__ob__ instanceof Observer) {
     ob = value.__ob__
   } else if (
     shouldObserve &&
     !isServerRendering() &&
     (Array.isArray(value) || isPlainObject(value)) &&
     Object.isExtensible(value) &&
     !value._isVue
   ) {
     ob = new Observer(value)
   }
   if (asRootData && ob) {
     ob.vmCount++
   }
   return ob
 }

Observer

遍历 data ，把数据变成响应式数据，分别对数组和对象采取不同的措施：

  • Object ： Object.defineProperty 自定义属性的 getter 和 setter
  • Array ：拦截/覆盖数组原型方法
 // Observer
 class Observer {
   constructor (value: any) {
     this.value = value
     // 收集依赖
     this.dep = new Dep()
     this.vmCount = 0
     // 在value上添加不可枚举的__ob__属性，并赋值为当前Observer实例
     def(value, '__ob__', this)
     if (Array.isArray(value)) {
       // 如果支持__proto__
       if ('__proto__' in {}) {
         // 设置value的原型为arrayMethods
         // arrayMethods是可以改变原数组的原型方法数组
         protoAugment(value, arrayMethods)
       } else {
         // 往value上添加不可枚举的数组方法 覆盖原型方法
         copyAugment(value, arrayMethods, arrayKeys)
       }
       // 遍历数组的每一项，对每一项都调用observe()
       this.observeArray(value)
     } else {
       // 观测对象的每个属性
       this.walk(value)
     }
   }
   
   
   walk (obj: Object) {  
     const keys = Object.keys(obj)
     for (let i = 0; i < keys.length; i++) {
       // 把对象的每个属性都变成响应式
       defineReactive(obj, keys[i])
     }
   }
 }

 function protoAugment (target, src: Object) {
   target.__proto__ = src
 }
 
 function copyAugment (target: Object, src: Object, keys: Array<string>) {
   for (let i = 0, l = keys.length; i < l; i++) {
     const key = keys[i]
     def(target, key, src[key])
   }
 }

defineReactive

定义属性的 getter 和 setter ：在 getter 中收集依赖，在 settet 中触发依赖更新

export function defineReactive (
   obj: Object,
   key: string,
   val: any,
   customSetter?: ?Function,
   shallow?: boolean
 ) {
   // 收集依赖
   // 和上面收集数组的依赖的位置是不同的
   const dep = new Dep()
   // 拿到对象属性的属性描述对象
   const property = Object.getOwnPropertyDescriptor(obj, key)
   // 如果该属性不能变化 直接return
   if (property && property.configurable === false) {
     return
   }
 
   // cater for pre-defined getter/setters
   const getter = property && property.get
   const setter = property && property.set
   // 如果只传了两个参数，比如在walk方法中调用，则赋值给val
   if ((!getter || setter) && arguments.length === 2) {
     val = obj[key]
   }
   // 观测obj对象key属性的值，如果值是一个对象则会递归调用，返回一个Observer实例
   let childOb = !shallow && observe(val)
   // 配置key属性的属性描述对象
   Object.defineProperty(obj, key, {
     enumerable: true,
     configurable: true,
     // 当获取obj.key的值得时候会触发get方法
     get: function reactiveGetter () {
       const value = getter ? getter.call(obj) : val
       // Dep.target为watcher实例
       if (Dep.target) {
         dep.depend()
         if (childOb) {
           childOb.dep.depend()
           if (Array.isArray(value)) {
             dependArray(value)
           }
         }
       }
       return value
     },
     // 当设置obj.key的值得时候会触发set方法
     set: function reactiveSetter (newVal) {
       // 获取本来的值
       const value = getter ? getter.call(obj) : val
       /* eslint-disable no-self-compare */
       if (newVal === value || (newVal !== newVal && value !== value)) {
         return
       }
       // #7981: for accessor properties without setter
       if (getter && !setter) return
       if (setter) {
         setter.call(obj, newVal)
       } else {
         val = newVal
       }
       // 观测新值得变化
       childOb = !shallow && observe(newVal)
       // 通知依赖进行更新
       dep.notify()
     }
   })
 }

arrayMethods 定义在 Observer/array.js 中：

const arrayProto = Array.prototype
export const arrayMethods = Object.create(arrayProto)

// 能够改变数组本身的方法
const methodsToPatch = [
  'push',
  'pop',
  'shift',
  'unshift',
  'splice',
  'sort',
  'reverse'
]

methodsToPatch.forEach(function (method) {
  // cache original method
  const original = arrayProto[method]
  // 在arrayMethods上定义methodsToPatch中的方法
  def(arrayMethods, method, function mutator (...args) {
    // 调用原数组原型方法
    const result = original.apply(this, args)
    // this是当前调用方法的数组，获取当前数组的__ob__属性也就是Observer实例
    const ob = this.__ob__
    let inserted
    switch (method) {
      case 'push':
      case 'unshift':
        inserted = args
        break
      case 'splice':
        inserted = args.slice(2)
        break
    }
    // 把新增的数据变成响应式数据
    if (inserted) ob.observeArray(inserted)
    // notify change 触发依赖更新
    ob.dep.notify()
    return result
  })
})

收集依赖和触发更新

到此为止响应式的数据已经生成了，下面就来看看怎么收集依赖和触发更新：

 class Dep {
   static target: ?Watcher;
   id: number;
   subs: Array<Watcher>;
 
   constructor () {
     this.id = uid++
     // 依赖数组
     this.subs = []
   }
 
   addSub (sub: Watcher) {
     this.subs.push(sub)
   }
 
   removeSub (sub: Watcher) {
     remove(this.subs, sub)
   }
 
   depend () {
     if (Dep.target) {
       Dep.target.addDep(this)
     }
   }
 
   notify () {
     const subs = this.subs.slice()
     for (let i = 0, l = subs.length; i < l; i++) {
       // 触发watcher的update方法更新dom
       subs[i].update()
     }
   }
 }
 Dep.target = null
 // 相当于一个栈
 const targetStack = []
 
 export function pushTarget (target: ?Watcher) {
   targetStack.push(target)
   Dep.target = target
 }
 
 export function popTarget () {
   targetStack.pop()
   Dep.target = targetStack[targetStack.length - 1]
 }

Watcher

在初始化数据后，就会生成 render 函数再真正的渲染到浏览器上。生成 render 函数后会利用 Watcher 来观测 render 函数的变化，这个 Watcher 称为 渲染watcher  。

 class Watcher {
     constructor (
     vm: Component,
     expOrFn: string | Function,
     cb: Function,
     options?: ?Object,
     isRenderWatcher?: boolean
   ) {
     this.vm = vm
     // 是否是渲染wathcer
     if (isRenderWatcher) {
       vm._watcher = this
     }
     vm._watchers.push(this)
     // options
     if (options) {
       this.deep = !!options.deep
       this.user = !!options.user
       this.lazy = !!options.lazy
       this.sync = !!options.sync
       this.before = options.before
     } else {
       this.deep = this.user = this.lazy = this.sync = false
     }
     this.cb = cb
     this.id = ++uid // uid for batching
     this.active = true
     this.dirty = this.lazy // for lazy watchers
     this.deps = []
     this.newDeps = []
     this.depIds = new Set()
     this.newDepIds = new Set()
     this.expression = process.env.NODE_ENV !== 'production'
       ? expOrFn.toString()
       : ''
     // parse expression for getter
     if (typeof expOrFn === 'function') {
       this.getter = expOrFn
     } else {
       this.getter = parsePath(expOrFn)
       if (!this.getter) {
         this.getter = noop
       }
     }
     // lazy是用于计算属性computed的其他的都是false，所以这里会执行get方法
     this.value = this.lazy
       ? undefined
       : this.get()
   }
   
   get () {
     // 设置Dep.target的值
     pushTarget(this)
     let value
     const vm = this.vm
     // 这里会执行defineReactive中定义的get方法，从而收集当前的watcher为依赖
     value = this.getter.call(vm, vm)
     if (this.deep) {
       traverse(value)
     }
     // 还原Dep.target的值
     popTarget()
     // 清理本轮没有渲染的watcher依赖
     this.cleanupDeps() 
     return value
   }
   // 添加依赖
   addDep (dep: Dep) {
     const id = dep.id
     if (!this.newDepIds.has(id)) {
       this.newDepIds.add(id)
       this.newDeps.push(dep)
       if (!this.depIds.has(id)) {
         dep.addSub(this)
       }
     }
   }
   // 清理本轮没有渲染的watcher依赖
   cleanupDeps () {
     let i = this.deps.length
     while (i--) {
       const dep = this.deps[i]
       if (!this.newDepIds.has(dep.id)) {
         dep.removeSub(this)
       }
     }
     let tmp = this.depIds
     this.depIds = this.newDepIds
     this.newDepIds = tmp
     this.newDepIds.clear()
     tmp = this.deps
     this.deps = this.newDeps
     this.newDeps = tmp
     this.newDeps.length = 0
   }
 }
