动态规划

动态规划（英语：Dynamic programming，简称DP）是一种在数学、管理科学、计算机科学、经济学和生物信息学中使用的，通过把原问题分解为相对简单的子问题的方式求解复杂问题的方法。 ——维基百科

具体例子

1. 连续子数组的最大和（剑指 Offer-42）


解法：

/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
  const len = nums.length
  if (!len) return 0
  let max = nums[0]
  let sum = nums[0]
  for (let i = 1; i < len; i++) {
    if (sum > 0) {
      sum += nums[i]
    } else {
      sum = nums[i]
    }
    max = Math.max(max, sum)
  }
  return max
};

2. 爬楼梯（LeetCode-70）


解法：

function climbStairs(n: number): number {
  let p: number = 0, q: number = 0, r: number = 1;
  for (let i = 1; i <= n; ++i) {
    p = q; 
    q = r; 
    r = p + q;
  }
  return r;
};

3. 使用最小花费爬楼梯


解法一：从前往后

// f[i] = cost[i] + min(f[i-1], f[i-2])
function minCostClimbingStairs(cost: number[]): number {
  let temp: number = 0
  // 前一个的花费
  let pre1:number = cost[1]
  // 前两个的花费
  let pre2: number = cost[0]
  for (let i = 2; i < cost.length; i++) {
    temp = Math.min(pre1, pre2) + cost[i]
    pre2 = pre1
    pre1 = temp
  }
  return Math.min(pre1, pre2)
};

4. 三步问题（面试题08.01）

function waysToStep(n: number): number {
  if (n === 1) return 1
  if (n === 2) return 2
  if (n === 3) return 4
  let ans:number = 0
  let i:number = 4
  // 前一位
  let pre1:number = 4
  // 前两位
  let pre2:number = 2
  // 前三位
  let pre3:number = 1
  while (i <= n) {
    ans = (pre1 + pre2 + pre3) % 1000000007
    pre3 = pre2
    pre2 = pre1
    pre1 = ans
    i++
  }
  return ans 
};

5. 打家劫舍（LeetCode-198）


解法一：

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function(nums) {
  const len = nums.length
  let result = [0]
  for (let i = 0; i < len; i++) {
    if (i >= 2) {
      result.push(Math.max.apply(null, result.slice(0, i)) + nums[i])
    } else {
      result.push(nums[i])
    }
  }
  return Math.max.apply(null, result)
};

var rob = function(nums) {
  const len = nums.length
  if (len < 3)return Math.max.apply(null, [...nums, 0])

  // 前两位
  let pre2 = nums[0]
  // 前一位
  let pre1 = nums[1]

    temp = pre2 + nums[i]
    // 保证pre2为pre2前面的最大值，因为并不是每次都是隔位相加，可能隔多位

  }
  return Math.max(pre1, pre2)
};

6. 连续数列（面试题16.17）


解法一：

/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
  const len = nums.length
  let result = []
  for (let i = 0; i < len; i++) {
    if (i >= 1) {
      
      result.push(Math.max(result[i - 1] + nums[i], nums[i]))
    } else {
      result.push(nums[i])
    }
  }
  return Math.max(...result)
};

  let ans = -Infinity

  for (let i = 0; i < len; i++) {

  }
  return ans
};

7. 买卖股票的最佳时机（LeetCode-121）


解法一：

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
  const len = prices.length
  let ans = 0
  prices.forEach((val, index) => {
    ans = Math.max(ans, Math.max.apply(null, prices.slice(index + 1)) - val)
  })
  return ans
};

var maxProfit = function(prices) {
  let min = prices[0]
  let res = 0
  prices.forEach(val => {
    res = Math.max(res, val - min)
    // 保证min是最小值
    min = Math.min(min, val)
  })
  return res
};

8. 除数博弈（LeetCode-1025）


解法一：

/**
 * @param {number} N
 * @return {boolean}
 */
var divisorGame = function(N) {
  let result = []
  result[1] = false
  result[2] = true
  for (let i = 3; i <= N; i++) {
    // 判断当 N = i时的胜负情况
    for (let j = 1; j < i; j++) {
      // 如果i % j === 0 并且当 N = i - j的时候，爱丽丝输了，则爱丽丝必胜
      if ( i % j === 0 && !result[i - j]) {
        result[i] = true
        break
      } else {
        result[i] = false
      }
    }
    
  }
  return result[N]
};
